    x: 0.5,
    y: 0.3,
    anchor: 'handoff-section'
  }

  type: EdgeType.INPUT,
  category: EdgeCategory.DEPENDENCY,

    section: 'imports',

    y: 0.1,
    anchor: 'imports-section'
  }
