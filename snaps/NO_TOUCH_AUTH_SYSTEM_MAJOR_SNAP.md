  constructor() {
    this.googleClient = new OAuth2Client(
      process.env.GOOGLE_CLIENT_ID,
      process.env.GOOGLE_CLIENT_SECRET,
      process.env.GOOGLE_REDIRECT_URI

    const ticket = await this.googleClient.verifyIdToken({
      idToken: googleIdToken,
      audience: process.env.GOOGLE_CLIENT_ID

    const payload = ticket.getPayload();
    
    // Extract user info (automatic)
    const userData = {
      email: payload.email!,
      name: payload.name!,
      firstName: payload.given_name,
      lastName: payload.family_name,
      avatar: payload.picture,
      emailVerified: payload.email_verified!,
      googleId: payload.sub
    };
    
    // Create or update user (automatic)
    const user = await this.createOrUpdateUser(userData, 'google');

  }

      success: true,
      message: `Check your email! We sent a magic link to ${email}`,
      magicLinkSent: true
    };
  }

   * Verify magic link and sign in

  async verifyMagicLink(token: string): Promise<User> {
    console.log('ğŸ”— Verifying magic link...');

    await this.invalidateMagicLinkToken(token);
    
    console.log(`âœ… Magic link verified: ${user.email}`);
