    start: number;

  /**
   * Start a story session for a user
   */
  startStory(userId: string, queenId: string, storyType: string): StorySession {
    const queen = this.queens.get(queenId);
    if (!queen) throw new Error(`Queen ${queenId} not found`);
    
    const session: StorySession = {
      sessionId: this.generateSessionId(),
      userId,
      queenId,
      storyType,
      currentScene: 0,
      startTime: new Date(),
      userChoices: [],
      progress: 0
    };
    
    this.activeStories.set(session.sessionId, session);
    
    // Begin the story
    this.playScene(session, queen.stories[storyType][0]);
    
    return session;
  }
  
  /**
   * Play a story scene with animation
   */
  private playScene(session: StorySession, scene: StoryScene): void {
    console.log(`\nğŸ QUEEN STORY: ${scene.title}`);
    
    // Narration
    scene.narration.forEach((line, index) => {
      setTimeout(() => {
        console.log(`   "${line}"`);
      }, index * 3000);
    });
    
    // Animation
    this.playAnimation(scene.animation);
    
    // Interactive elements
    if (scene.interactive_elements) {
      this.setupInteractivity(session, scene.interactive_elements);
    }
    
    // Auto-advance or wait for user
    if (!scene.next_scenes || scene.next_scenes.length === 1) {
      setTimeout(() => {
        this.advanceStory(session, scene.next_scenes[0]);
      }, scene.duration * 1000);
    }
  }
  
  /**
   * Get all queens in a shell
   */
  getQueensByShell(shell: ShellNumber): QueenRoleStory[] {
    return Array.from(this.queens.values())
      .filter(queen => queen.shell === shell);
  }
  
  /**
   * Get recommended story for a user action
   */
  getRelevantStory(action: string): { queenId: string; storyType: string } | null {
    // Map user actions to relevant queen stories
    const storyMap = {
      'signup': { queenId: 'queen-auth-01', storyType: 'user_interaction' },
      'transaction': { queenId: 'queen-vibechain-01', storyType: 'daily_work' },
