// Default landing for ALL users
const LANDING_PAGE = '/man-cave';

// Landing experience by tier
function getLandingExperience(tier: MembershipTier): LandingExperience {
  return {
    location: 'Count of Monte Cristo Man Cave',
    setting: 'At the Resort',
    
    welcome: getWelcomeMessage(tier),
    access: getAccessLevel(tier),
    interactive: getInteractionPoints(tier),
    bbheLevel: getBBHELevel(tier),
    
    navigation: {
      library: '/docs',           // All tiers
      display: '/products',        // All tiers
      seating: '/experiences',     // All tiers
      windows: '/resort',          // View level by tier
      concierge: '/support',       // All tiers
      privateDoor: tier >= MembershipTier.MEMBERS_ONLY ? '/members' : null,
      elTaino: tier === MembershipTier.ULTIMATE_VIP ? '/drive' : '/view-el-taino'
    },
    
    callToAction: getCallToAction(tier)
  };
}

function getWelcomeMessage(tier: MembershipTier): string {
  switch(tier) {
    case MembershipTier.VISITOR:
      return "Step into the man cave (preview mode)";
    case MembershipTier.GUEST:
      return "Welcome back! Your purchase awaits";
    case MembershipTier.MEMBERS_ONLY:
      return "Welcome home, member";
    case MembershipTier.BACKSTAGE_PASS:
      return "Welcome to the creator's space";
    case MembershipTier.ULTIMATE_VIP:
      return "Welcome to your private quarters";
  }
}

// Always start here
app.get('/', (req, res) => {
  const userTier = getUserTier(req.user);
  const experience = getLandingExperience(userTier);
  res.render('man-cave-landing', experience);
});
