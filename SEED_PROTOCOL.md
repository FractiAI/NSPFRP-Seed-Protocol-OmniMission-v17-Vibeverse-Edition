class OmniMissionAttention(nn.Module):
    def __init__(self, num_domains, hidden_dim):
        self.domain_embeddings = nn.Embedding(num_domains, hidden_dim)
        self.attention_heads = nn.MultiheadAttention(hidden_dim, num_heads=8)
