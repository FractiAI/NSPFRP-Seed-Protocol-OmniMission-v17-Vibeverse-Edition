```typescript
interface NSPFRNPAddress {
  // Complete address string
  fullAddress: string;
  
  // Layered components
  layers: {
    layer0_root?: string;        // e.g., "root.nspfrnp"
    layer1_trunk?: string;       // e.g., "trunk.fsr-system"
    layer2_branch?: string;      // e.g., "branch.experience"
    layer3_sub?: string;         // e.g., "sub.eternity-mode"
    layer4_leaf?: string;        // e.g., "leaf.fsr-shell-emission"
    layer5_node?: string;        // e.g., "node.hhf-fb-002"
  };
  
  // Metadata
  octave: number;                // e.g., 7.75
  tier?: number;                 // 1, 2, or 3
  nodeType?: string;             // fruiting-body, hyphae, mycelium, spore, edge
  
  // Recursion trace
  ancestry: string[];            // Complete path from root to current
  parent?: string;               // Immediate parent address
  children?: string[];           // Direct children addresses
  siblings?: string[];           // Sibling addresses
  
  // Position
  position?: {x: number, y: number, z?: number};
  
  // Additional metadata
  capacity?: NodeCapacity;
  device?: string;
  user?: string;
  location?: string;
  connection?: string;
  bandwidth?: string;
}

class NSPFRNPAddressing {
  /**
   * Parse full address into components
   */
  parse(address: string): NSPFRNPAddress {
    const parts = address.split('/').filter(p => p);
    
    const layers: any = {};
    let octave = 0;
    let tier: number | undefined;
    let nodeType: string | undefined;
    
    // Parse each component
    for (const part of parts) {
      const [type, name] = part.split('.');
      
      if (type === 'root') layers.layer0_root = part;
      else if (type === 'trunk') layers.layer1_trunk = part;
      else if (type === 'branch') layers.layer2_branch = part;
      else if (type === 'sub') layers.layer3_sub = part;
      else if (type === 'leaf') layers.layer4_leaf = part;
      else if (type === 'node') layers.layer5_node = part;
      else if (type === 'octave') octave = parseFloat(name);
      else if (type === 'tier') tier = parseInt(name);
      else if (type === 'type') nodeType = name;
    }
    
    return {
      fullAddress: address,
      layers,
      octave,
      tier,
      nodeType,
      ancestry: this.getAncestry(address),
      parent: this.getParent(address),
      children: [], // Would query from database
      siblings: []  // Would query from database
    };
  }
  
  /**
   * Build address from components
   */
  build(components: Partial<NSPFRNPAddress>): string {
    const parts: string[] = [];
    
    const { layers, octave, tier, nodeType, capacity, position } = components;
    
    if (layers?.layer0_root) parts.push(layers.layer0_root);
    if (layers?.layer1_trunk) parts.push(layers.layer1_trunk);
    if (layers?.layer2_branch) parts.push(layers.layer2_branch);
    if (layers?.layer3_sub) parts.push(layers.layer3_sub);
    if (layers?.layer4_leaf) parts.push(layers.layer4_leaf);
    if (layers?.layer5_node) parts.push(layers.layer5_node);
    if (octave !== undefined) parts.push(`octave.${octave}`);
    if (tier !== undefined) parts.push(`tier.${tier}`);
    if (nodeType) parts.push(`type.${nodeType}`);
    if (capacity) parts.push(`capacity.${capacity.compute}-${capacity.storage}-${capacity.network}-${capacity.grammar}`);
    if (position) parts.push(`position.${position.x}-${position.y}-${position.z || 0}`);
    
    return '/' + parts.join('/');
  }
  
  /**
   * Get ancestry chain
   */
  getAncestry(address: string): string[] {
    const parts = address.split('/').filter(p => p);
    const ancestry: string[] = [];
    
    for (let i = 1; i <= parts.length; i++) {
      ancestry.push('/' + parts.slice(0, i).join('/'));
    }
    
    return ancestry;
```
