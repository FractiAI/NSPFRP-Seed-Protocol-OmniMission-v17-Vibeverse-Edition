// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract OctaveActivationSystem {
    
    struct NodeSubscription {
        uint256 nodeId;
        uint8 highestActivatedOctave;
        uint8 currentActiveOctave;
        uint256 nextPaymentDue;
        bool isPaused;
        uint256 lifetimeRevenue;
    }
    
    // Goldilocks activation key prices (ONE perfect price per octave)
    uint256[9] public activationKeyPrices = [
        1e18,          // Octave 0: $1 (node purchase)
        10e18,         // Octave 1: $10
        50e18,         // Octave 2: $50
        250e18,        // Octave 3: $250
        1000e18,       // Octave 4: $1,000
        5000e18,       // Octave 5: $5,000
        25000e18,      // Octave 6: $25,000
        100000e18,     // Octave 7: $100,000
        1000000e18     // Octave 8: $1,000,000
    ];
    
    // Goldilocks monthly fees (ONE perfect price per octave)
    uint256[9] public monthlyFees = [
        0,             // Octave 0: $0
        5e18,          // Octave 1: $5
        20e18,         // Octave 2: $20
        100e18,        // Octave 3: $100
        500e18,        // Octave 4: $500
        2500e18,       // Octave 5: $2,500
        10000e18,      // Octave 6: $10,000
        50000e18,      // Octave 7: $50,000
        250000e18      // Octave 8: $250,000
    ];
    
    mapping(uint256 => NodeSubscription) public subscriptions;
    mapping(uint256 => mapping(uint8 => bool)) public octaveActivations;
    
    event OctaveActivated(uint256 indexed nodeId, uint8 octave, uint256 price);
    event MonthlyFeePaid(uint256 indexed nodeId, uint8 octave, uint256 amount);
    event OctaveChanged(uint256 indexed nodeId, uint8 oldOctave, uint8 newOctave);
    event AccountPaused(uint256 indexed nodeId);
    event AccountResumed(uint256 indexed nodeId);
    
    // Purchase node (Octave 0)
    function purchaseNode(uint256 nodeId) external payable {
        require(msg.value >= activationKeyPrices[0], "Insufficient payment for node");
        require(subscriptions[nodeId].nodeId == 0, "Node already exists");
        
        subscriptions[nodeId] = NodeSubscription({
            nodeId: nodeId,
            highestActivatedOctave: 0,
            currentActiveOctave: 0,
            nextPaymentDue: block.timestamp,
            isPaused: false,
            lifetimeRevenue: msg.value
        });
        
        octaveActivations[nodeId][0] = true;
        emit OctaveActivated(nodeId, 0, msg.value);
    }
    
    // Activate octave (one-time key purchase)
    function activateOctave(uint256 nodeId, uint8 octave) external payable {
        require(octave > 0 && octave <= 8, "Invalid octave");
        require(subscriptions[nodeId].nodeId != 0, "Node does not exist");
        require(!octaveActivations[nodeId][octave], "Octave already activated");
        require(octaveActivations[nodeId][octave - 1], "Must activate previous octave first");
        require(msg.value >= activationKeyPrices[octave], "Insufficient payment for activation");
        
        octaveActivations[nodeId][octave] = true;
        if (octave > subscriptions[nodeId].highestActivatedOctave) {
            subscriptions[nodeId].highestActivatedOctave = octave;
        }
        subscriptions[nodeId].lifetimeRevenue += msg.value;
        
        emit OctaveActivated(nodeId, octave, msg.value);
    }
    
    // Pay monthly fee
    function payMonthlyFee(uint256 nodeId) external payable {
        NodeSubscription storage sub = subscriptions[nodeId];
        require(sub.nodeId != 0, "Node does not exist");
        require(!sub.isPaused, "Account is paused");
        require(block.timestamp >= sub.nextPaymentDue, "Payment not yet due");
        
        uint8 octave = sub.currentActiveOctave;
        require(msg.value >= monthlyFees[octave], "Insufficient monthly fee");
        
        sub.nextPaymentDue = block.timestamp + 30 days;
        sub.lifetimeRevenue += msg.value;
        
        emit MonthlyFeePaid(nodeId, octave, msg.value);
    }
    
    // Upgrade to higher octave
    function upgradeOctave(uint256 nodeId, uint8 newOctave) external {
        NodeSubscription storage sub = subscriptions[nodeId];
        require(sub.nodeId != 0, "Node does not exist");
        require(octaveActivations[nodeId][newOctave], "Octave not activated");
        require(newOctave > sub.currentActiveOctave, "Can only upgrade");
        
        uint8 oldOctave = sub.currentActiveOctave;
        sub.currentActiveOctave = newOctave;
        
        emit OctaveChanged(nodeId, oldOctave, newOctave);
    }
    
    // Downgrade to lower octave
    function downgradeOctave(uint256 nodeId, uint8 newOctave) external {
        NodeSubscription storage sub = subscriptions[nodeId];
        require(sub.nodeId != 0, "Node does not exist");
        require(octaveActivations[nodeId][newOctave], "Octave not activated");
        require(newOctave < sub.currentActiveOctave, "Can only downgrade");
        
        uint8 oldOctave = sub.currentActiveOctave;
        sub.currentActiveOctave = newOctave;
        
        emit OctaveChanged(nodeId, oldOctave, newOctave);
    }
    
    // Pause account
    function pauseAccount(uint256 nodeId) external {
        NodeSubscription storage sub = subscriptions[nodeId];
        require(sub.nodeId != 0, "Node does not exist");
        require(!sub.isPaused, "Account already paused");
        
        sub.isPaused = true;
        emit AccountPaused(nodeId);
    }
    
    // Resume account
    function resumeAccount(uint256 nodeId) external payable {
        NodeSubscription storage sub = subscriptions[nodeId];
        require(sub.nodeId != 0, "Node does not exist");
        require(sub.isPaused, "Account not paused");
        require(msg.value >= monthlyFees[sub.currentActiveOctave], "Pay monthly fee to resume");
        
        sub.isPaused = false;
        sub.nextPaymentDue = block.timestamp + 30 days;
        sub.lifetimeRevenue += msg.value;
        
        emit AccountResumed(nodeId);
    }
    
    // Check if node has access to octave
    function hasOctaveAccess(uint256 nodeId, uint8 octave) external view returns (bool) {
        NodeSubscription storage sub = subscriptions[nodeId];
        if (sub.isPaused) return false;
        if (block.timestamp >= sub.nextPaymentDue) return false; // Payment overdue
        if (!octaveActivations[nodeId][octave]) return false; // Not activated
        if (octave > sub.currentActiveOctave) return false; // Higher than current
        return true;
    }
    
    // Get lifetime revenue from node
    function getLifetimeRevenue(uint256 nodeId) external view returns (uint256) {
        return subscriptions[nodeId].lifetimeRevenue;
    }
}
