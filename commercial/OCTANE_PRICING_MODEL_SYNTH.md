contract OctaneActivationSystem is Ownable, ReentrancyGuard {
    
    IERC20 public synthToken; // SYNTH ERC-20 token contract
    
    struct NodeSubscription {
        uint256 nodeId;
        uint8 highestActivatedOctane;
        uint8 currentActiveOctane;
        uint256 nextPaymentDue;
        bool isPaused;
        uint256 lifetimeRevenueSynth; // Tracked in SYNTH tokens
    }
    
    // Goldilocks activation key prices (ONE perfect price per OCTANE, in SYNTH)
    // 1 SYNTH = 1e18 wei (standard ERC-20 decimals)
    uint256[9] public activationKeyPricesSynth = [
        1e18,           // OCTANE 0: 1 SYNTH (node purchase)
        10e18,          // OCTANE 1: 10 SYNTH
        50e18,          // OCTANE 2: 50 SYNTH
        250e18,         // OCTANE 3: 250 SYNTH
        1000e18,        // OCTANE 4: 1,000 SYNTH
        5000e18,        // OCTANE 5: 5,000 SYNTH
        25000e18,       // OCTANE 6: 25,000 SYNTH
        100000e18,      // OCTANE 7: 100,000 SYNTH
        1000000e18      // OCTANE 8: 1,000,000 SYNTH
    ];
    
    // Goldilocks monthly fees (ONE perfect price per OCTANE, in SYNTH)
    uint256[9] public monthlyFeesSynth = [
        0,              // OCTANE 0: 0 SYNTH
        5e18,           // OCTANE 1: 5 SYNTH
        20e18,          // OCTANE 2: 20 SYNTH
        100e18,         // OCTANE 3: 100 SYNTH
        500e18,         // OCTANE 4: 500 SYNTH
        2500e18,        // OCTANE 5: 2,500 SYNTH
        10000e18,       // OCTANE 6: 10,000 SYNTH
        50000e18,       // OCTANE 7: 50,000 SYNTH
        250000e18       // OCTANE 8: 250,000 SYNTH
    ];
    
    mapping(uint256 => NodeSubscription) public subscriptions;
    mapping(uint256 => mapping(uint8 => bool)) public octaneActivations;
    
    event OctaneActivated(uint256 indexed nodeId, uint8 octane, uint256 priceSynth);
    event MonthlyFeePaid(uint256 indexed nodeId, uint8 octane, uint256 amountSynth);
    event OctaneChanged(uint256 indexed nodeId, uint8 oldOctane, uint8 newOctane);
    event AccountPaused(uint256 indexed nodeId);
    event AccountResumed(uint256 indexed nodeId);
    
    constructor(address _synthTokenAddress) {
        synthToken = IERC20(_synthTokenAddress);
    }
    
    // Purchase node (OCTANE 0) - pay in SYNTH
    function purchaseNode(uint256 nodeId) external nonReentrant {
        require(subscriptions[nodeId].nodeId == 0, "Node already exists");
        require(synthToken.transferFrom(msg.sender, address(this), activationKeyPricesSynth[0]), "SYNTH transfer failed");
        
        subscriptions[nodeId] = NodeSubscription({
            nodeId: nodeId,
            highestActivatedOctane: 0,
            currentActiveOctane: 0,
            nextPaymentDue: block.timestamp,
            isPaused: false,
            lifetimeRevenueSynth: activationKeyPricesSynth[0]
        });
        
        octaneActivations[nodeId][0] = true;
        emit OctaneActivated(nodeId, 0, activationKeyPricesSynth[0]);
    }
    
    // Activate OCTANE (one-time key purchase) - pay in SYNTH
    function activateOctane(uint256 nodeId, uint8 octane) external nonReentrant {
        require(octane > 0 && octane <= 8, "Invalid octane");
        require(subscriptions[nodeId].nodeId != 0, "Node does not exist");
        require(!octaneActivations[nodeId][octane], "Octane already activated");
        require(octaneActivations[nodeId][octane - 1], "Must activate previous octane first");
        require(synthToken.transferFrom(msg.sender, address(this), activationKeyPricesSynth[octane]), "SYNTH transfer failed");
        
        octaneActivations[nodeId][octane] = true;
        if (octane > subscriptions[nodeId].highestActivatedOctane) {
            subscriptions[nodeId].highestActivatedOctane = octane;
        }
        subscriptions[nodeId].lifetimeRevenueSynth += activationKeyPricesSynth[octane];
        
        emit OctaneActivated(nodeId, octane, activationKeyPricesSynth[octane]);
    }
    
    // Pay monthly fee - pay in SYNTH
    function payMonthlyFee(uint256 nodeId) external nonReentrant {
        NodeSubscription storage sub = subscriptions[nodeId];
        require(sub.nodeId != 0, "Node does not exist");
        require(!sub.isPaused, "Account is paused");
        require(block.timestamp >= sub.nextPaymentDue, "Payment not yet due");
        
        uint8 octane = sub.currentActiveOctane;
        require(synthToken.transferFrom(msg.sender, address(this), monthlyFeesSynth[octane]), "SYNTH transfer failed");
        
        sub.nextPaymentDue = block.timestamp + 30 days;
        sub.lifetimeRevenueSynth += monthlyFeesSynth[octane];
        
        emit MonthlyFeePaid(nodeId, octane, monthlyFeesSynth[octane]);
    }
    
    // Upgrade to higher OCTANE
    function upgradeOctane(uint256 nodeId, uint8 newOctane) external {
        NodeSubscription storage sub = subscriptions[nodeId];
        require(sub.nodeId != 0, "Node does not exist");
        require(octaneActivations[nodeId][newOctane], "Octane not activated");
        require(newOctane > sub.currentActiveOctane, "Can only upgrade");
        
        uint8 oldOctane = sub.currentActiveOctane;
        sub.currentActiveOctane = newOctane;
        
        emit OctaneChanged(nodeId, oldOctane, newOctane);
    }
    
    // Downgrade to lower OCTANE
    function downgradeOctane(uint256 nodeId, uint8 newOctane) external {
        NodeSubscription storage sub = subscriptions[nodeId];
        require(sub.nodeId != 0, "Node does not exist");
        require(octaneActivations[nodeId][newOctane], "Octane not activated");
        require(newOctane < sub.currentActiveOctane, "Can only downgrade");
        
        uint8 oldOctane = sub.currentActiveOctane;
        sub.currentActiveOctane = newOctane;
        
        emit OctaneChanged(nodeId, oldOctane, newOctane);
    }
    
    // Pause account (stop monthly fees)
    function pauseAccount(uint256 nodeId) external {
        NodeSubscription storage sub = subscriptions[nodeId];
        require(sub.nodeId != 0, "Node does not exist");
        require(!sub.isPaused, "Account already paused");
        
        sub.isPaused = true;
        emit AccountPaused(nodeId);
    }
    
    // Resume account (pay monthly fee to restart)
    function resumeAccount(uint256 nodeId) external nonReentrant {
        NodeSubscription storage sub = subscriptions[nodeId];
        require(sub.nodeId != 0, "Node does not exist");
        require(sub.isPaused, "Account not paused");
        require(synthToken.transferFrom(msg.sender, address(this), monthlyFeesSynth[sub.currentActiveOctane]), "SYNTH transfer failed");
        
        sub.isPaused = false;
        sub.nextPaymentDue = block.timestamp + 30 days;
        sub.lifetimeRevenueSynth += monthlyFeesSynth[sub.currentActiveOctane];
        
        emit AccountResumed(nodeId);
    }
    
    // Check if node has access to OCTANE
    function hasOctaneAccess(uint256 nodeId, uint8 octane) external view returns (bool) {
        NodeSubscription storage sub = subscriptions[nodeId];
        if (sub.isPaused) return false;
        if (block.timestamp >= sub.nextPaymentDue) return false; // Payment overdue
        if (!octaneActivations[nodeId][octane]) return false; // Not activated
        if (octane > sub.currentActiveOctane) return false; // Higher than current
        return true;
    }
    
    // Get lifetime revenue from node (in SYNTH)
    function getLifetimeRevenueSynth(uint256 nodeId) external view returns (uint256) {
        return subscriptions[nodeId].lifetimeRevenueSynth;
    }
    
    // Withdraw SYNTH (owner only, for treasury)
    function withdrawSynth(uint256 amount) external onlyOwner {
        require(synthToken.transfer(owner(), amount), "SYNTH transfer failed");
    }
}
