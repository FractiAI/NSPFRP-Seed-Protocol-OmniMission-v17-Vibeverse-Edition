## ðŸŽ¯ Usage Example

### Create Invitation for Call Participants

```typescript
import { InvitationManager } from './src/invitation/invitation-manager.js';

const manager = new InvitationManager();

const participants = [
  { id: '1', name: 'Alice', email: 'alice@example.com', selected: true },
  { id: '2', name: 'Bob', phone: '+1234567890', selected: true },
  { id: '3', name: 'Charlie', email: 'charlie@example.com', selected: false }
];

const package = manager.createInvitationPackage(participants);

// Send emails
package.emails.forEach(email => {
  // Send via email service
  sendEmail(email.to, email.subject, email.html);
});

// Send messages
package.messages.forEach(message => {
  // Send via messaging service
  sendMessage(message.to, message.message);
});

// Share button
console.log(package.buttons.html); // HTML button
console.log(package.buttons.markdown); // Markdown button
```
